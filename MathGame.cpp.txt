#include <iostream>
using namespace std;
int Roundem(int form, int to) {
	int Roun = rand() % (to - form + 1) + form;
	return Roun;
}
char GetRandomOperator() {
	int r = rand() % 4;
	switch (r) {
	case 0: return '+';
	case 1: return '-';
	case 2: return '*';
	case 3: return '/';
	}
	return '+'; 
}

	if (Level == 1) {

		Num1 = Roundem(10, 50);

}

bool CheckYesOrNo() {
	char ch;
	cout << "Are You Play Agen(Y/N): \n";
	cin >> ch;
	if (ch == 'Y' || ch == 'y') {
		return true;
	}
	else if (ch == 'N' || ch == 'n') {
		return false;
	}
	else {
		cout << "Enter Y or N\n";
		return CheckYesOrNo();
	}
}

void MahtGame(short &Num1, short &Num2, short &Level) {

		cout << "The Type Of Question(+,-,/,*) or Mix[M]: \n";
